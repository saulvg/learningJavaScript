          function adjustLights(lights) {
            let changesStartingGreen = 0;
            let changesStartingRed = 0;
        
            for (let i = 0; i < lights.length; i++) {
                const startGreen = i % 2 === 0 ? '游릭' : '游댮';
                const startRed = i % 2 === 0 ? '游댮' : '游릭';
              
                // Contabilizar cambios para comenzar con luces verdes
                if (lights[i] !== startGreen) {
                    changesStartingGreen++;
                }
        
                // Contabilizar cambios para comenzar con luces rojas
                if (lights[i] !== startRed) {
                    changesStartingRed++;
                }
            }
        
            // Devolver el menor n칰mero de cambios
            return Math.min(changesStartingGreen, changesStartingRed);

          function adjustLights(lights) {
            let changesStartingGreen = 0;
        
            // Contabilizar cambios para comenzar con luces verdes
            for (let i = 0; i < lights.length; i++) {
                const expectedColor = i % 2 === 0 ? '游릭' : '游댮';
                if (lights[i] !== expectedColor) {
                    changesStartingGreen++;
                }
            }
        
            // Obtener el total de luces
            const totalLights = lights.length;
        
            // Para obtener los cambios para comenzar con luces rojas, invertimos los cambios
            const changesStartingRed = totalLights - changesStartingGreen;
          
        
            // Devolver el menor n칰mero de cambios
